#!/bin/bash
yum update -y
yum install httpd php php-mysqli stress -y
yum install mariadb105
cd /etc/httpd/conf
cp httpd.conf httpdconfbackup.conf
rm -rf httpd.conf
wget https://htaccess-bucket.s3.ap-south-1.amazonaws.com/httpd.conf
cd /var/www/html
wget https://wordpress.org/latest.tar.gz
tar -xzf latest.tar.gz
cp -rf wordpress/* /var/www/html
rm -rf wordpress
rm -rf latest.tar.gz
chmod -R 755 wp-content
chown -R apache.apache wp-content
service httpd start
chkconfig httpd on
